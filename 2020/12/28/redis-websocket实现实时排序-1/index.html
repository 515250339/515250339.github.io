views.py

from dwebsocket import accept_websocket


# 获取排行榜
@accept_websocket
def ranking(request):
    if request.is_websocket():
        while 1:
            time.sleep(1)  ## 向前端发送时间
            try:
                list2 = r6.zrange("ranking", 0, 10, withscores=True)
                list2.reverse()
                list3 = []
                # print(list2)
                s = 0
                for i in list2:
                    s += 1
                    list3.append({"name": i[0].decode(), "total": i[1], "ranking": s})
                request.websocket.send(json.dumps(list3))
            except:
                request.websocket.send(json.dumps([{"message": ""}]))
